      <button onclick="moveSprite('left')">←</button>
      <button onclick="moveSprite('up')">↑</button>
      <button onclick="moveSprite('down')">↓</button>
      <button onclick="moveSprite('right')">→</button>
    </div>

  <script>
    const sprite = document.getElementById('sprite');
    let x = 10, y = 10;

    function moveSprite(direction) {
      switch (direction) {
        case 'left': x -= 10; break;
        case 'right': x += 10; break;
        case 'up': y -= 10; break;
        case 'down': y += 10; break;
      }
      sprite.style.left = x + 'px';
      sprite.style.top = y + 'px';
    }
  </script>
